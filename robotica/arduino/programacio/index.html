// 02_01_blink

void setup() { //Codi que s'execita una vegada i és la onfiguració.
  pinMode(13, OUTPUT); //Estem dient que el pin 13 és una sortida de corrent o OUTPUT on puc connectar
}

void loop() {  //ës un codi que s'executa sense parar fins que desconnecto l'Arduino o li dono una instrucció de parada.
 digitalWrite(13, HIGH);  //digitalWrite és ua sortida digital i permet encendre i apagar. Només té 22 valors, HIGH i LOW (Encès i apagat).
 delay(500);  //És el temps en milisegans que ha d'estar, en aquet cas encès
 digitalWrite(13, LOW);  //Aqui digitalWrite fa que s'apagui
 delay(500);
}
